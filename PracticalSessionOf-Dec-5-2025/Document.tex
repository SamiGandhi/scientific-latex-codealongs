class Car:
    def __init__(self, color, price):
        self.color = color
        self.price = price

    def getColor(self):
        return self.color

    def getPrice(self):
        return self.price

    def setPrice(self, price):
        self.price = price

car1 = Car("Red", 15000)
print(car1.getColor())
print(car1.getPrice())
